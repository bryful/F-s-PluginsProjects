//-------------------------------------------------------------------------------------------------
static PF_Err
GetParams(CFsAE* ae, FillInfo* infoP)
{
	PF_Err		err = PF_Err_NONE;

	ERR(ae->GetADD(ID_Color_COUNT, &infoP->target_count));
	for (int i = 0; i < 10; i++) {
		ERR(ae->GetCOLOR(ID_Color1 + i, &infoP->targets[i]));
	}
	ERR(ae->GetCHECKBOX(ID_KEEP_PIXELS, &infoP->KeepPixel));
	ERR(ae->GetCHECKBOX(ID_FillUnknownColors, &infoP->FillUnknownColors));

	PF_Pixel8 c;
	ERR(ae->GetCOLOR(ID_Fill_Color, &c));
	infoP->GiveUpColor8 = c;
	infoP->GiveUpColor16 = CONV8TO16(c);
	infoP->GiveUpColor32 = CONV8TO32(c);

	ERR(ae->GetCHECKBOX(ID_MakeWhiteTransparent, &infoP->MakeWhiteTransparent));

	return err;
}
//-------------------------------------------------------------------------------------------------
static PF_Err
Exec(CFsAE* ae, FillInfo* infoP)
{
	PF_Err	err = PF_Err_NONE;

	//実際の処理は別ファイル（CellLineEraserSub.cpp）に記述
	err = CellLineEraserSub(ae, infoP);

	return err;
}

//-------------------------------------------------------------------------------------------------
//メインレンダリング関数
static PF_Err
Render(
	PF_InData* in_data,
	PF_OutData* out_data,
	PF_ParamDef* params[],
	PF_LayerDef* output)
{
	PF_Err	err = PF_Err_NONE;
	CFsAE ae(in_data, out_data, params, output, ID_NUM_PARAMS);
	err = ae.resultErr();
	if (!err) {
		FillInfo info;
		ERR(GetParams(&ae, &info));
		ERR(Exec(&ae, &info));
	}
	return err;
}

/*
	SmartFX対応の場合、まずこの関数が呼ばれてパラメータの獲得を行う
*/
#if defined(SUPPORT_SMARTFX)
static PF_Err
PreRender(
	PF_InData* in_data,
	PF_OutData* out_data,
	PF_PreRenderExtra* extraP)
{
	PF_Err		err = PF_Err_NONE;
	CFsAE ae(in_data, out_data, extraP, sizeof(FillInfo), ID_NUM_PARAMS);
	err = ae.resultErr();
	if (!err) {

		FillInfo* infoP = reinterpret_cast<FillInfo*>(ae.LockPreRenderData());
		if (infoP) {
			ae.SetHostPreRenderData();
			ERR(GetParams(&ae, infoP));
			ERR(ae.UnSetPreRenderData());
			ae.UnlockPreRenderData();
		}
		else {
			err = PF_Err_OUT_OF_MEMORY;
		}
	}
	return err;
}
#endif
//-----------------------------------------------------------------------------------
#if defined(SUPPORT_SMARTFX)
static PF_Err
SmartRender(
	PF_InData* in_data,
	PF_OutData* out_data,
	PF_SmartRenderExtra* extraP)
{
	PF_Err			err = PF_Err_NONE;
	CFsAE ae(in_data, out_data, extraP, sizeof(FillInfo), ID_NUM_PARAMS);
	err = ae.resultErr();
	if (!err) {
		FillInfo* infoP = reinterpret_cast<FillInfo*>(ae.GetPreRenderData());
		if (infoP) {
			ERR(Exec(&ae, infoP));
		}
		else {
			err = PF_Err_OUT_OF_MEMORY;
		}
	}
	return err;
}
#endif
